SELECT (Embedded SQL)

Use this command to retrieve data from one or more tables, views, or snapshots, assigning the selected values to host variables. 

 EXEC SQL
   [ AT { dbname | :host_variable } ]
   SELECT select_list
 INTO
   :host_variable [ [ INDICATOR ] :indicator_variable ]
   [, :host_variable [ [ INDICATOR ] :indicator_variable ] ] ...
 FROM table _list
 [ WHERE condition ]
 [ [ START WITH condition ] CONNECT BY condition
   | GROUP BY expr [, expr ] ... [ HAVING condition ] ] ...
 [ { UNION
   | UNION ALL
   | INTERSECT
   | MINUS } SELECT command ]
 [ ORDER BY
   { expr | position } [ ASC | DESC ]
   [, { expr | position } [ ASC | DESC ] ] ...
   | FOR UPDATE
     [ OF [ [ schema. ]{ table. | view. } ] column ]
     [, OF [ [ schema. ]{ table. | view. } ] column ] ...
     [ NOWAIT ]  ... ]
